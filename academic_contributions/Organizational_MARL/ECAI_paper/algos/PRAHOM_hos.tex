    \caption{PRAHOM-hos}\label{alg:PRAHOM-hos}

    \KwData{$sh_{joint,i,s}$, joint-histories to be analyzed} \Comment{PRAHOM context}

    \KwData{...}

    \KwResult{$os \subseteq OS$, associated organization specifications}

    \ForEach{$h_{joint,i} \in SH_{joint,i,s}$}{

        \Comment{KOSI: identify from known relations}

    }

    \uIf{any org. spec. $\in os$ is $\emptyset$}{

        \Comment{GOSI: infer from OS definitions}

        $os = GOSI(os, h_{joint,i},\dots)$

    }
