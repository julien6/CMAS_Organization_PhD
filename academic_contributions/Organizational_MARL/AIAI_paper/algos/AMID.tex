
\RestyleAlgo{ruled}
\SetKwComment{Comment}{// }{}

\begin{algorithm}[hbt!]
    \caption{Actions based $\mathcal{M}OISE^+$ Identification (AMID)}\label{alg:amid}
    \KwData{$\pi_{joint}$, the joint-policies to be analyzed}
    \KwResult{$os \in OS$, the associated organization specifications}
    
    % Set Function Names
    \SetKwFunction{FMain}{Main}
    \SetKwFunction{FDetermineRoles}{DetermineRoles}
    \SetKwProg{Fn}{Function}{:}{}
    
    \ForEach{$\pi_{joint,i} \in \pi_{joint}$}{
        \ForEach{$\pi_{i} \in \pi_{joint,i}$}{
        
            $\mathcal{A}_{i,j} := \{a_k \in A | (\omega_k,a_k) \in \pi_{i}\}$  \Comment*[r]{Played actions}
            
            $\mathcal{R}_{i,j} := DetermineRoles(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineSubGroups(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineIntraLinks(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineInterLinks(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineIntraCompatibilities(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineInterCompatibilities(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineNp(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineNg(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineGoals(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineMissions(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DeterminePlans(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineNo(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineNm(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DeterminePreferenceOrder(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DetermineObligations(\mathcal{A}_{i,j})$
            
            $\mathcal{R}_{i,j} := DeterminePermissions(\mathcal{A}_{i,j})$
            
        }
    }
\end{algorithm}
