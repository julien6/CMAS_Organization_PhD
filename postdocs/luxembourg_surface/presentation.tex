\section{Problem Framing}
\begin{frame}{Problem framing: Current analyst workflow remains tedious}
    \begin{itemize}
        \item Attack surface analysis still means switching across many tools.
        \item Outputs are heterogeneous and mostly textual.
        \item Analysts repeatedly correlate findings, tune configs, and relaunch scans.
        \item Reproducibility and traceability are often weak across runs.
    \end{itemize}

    \vspace{0.4em}
    \textbf{Target:} automate the repetitive loop, keep analyst control and final validation.
    \note{This is the practical gap we are trying to close: less manual orchestration, better traceability, faster triage.}
\end{frame}

\begin{frame}{What we want to build (Concrete target system)}
    \begin{itemize}
        \item \textbf{DRL orchestrator} to pick next tool and parameters.
        \item \textbf{Unified interface/language} across EASM tool outputs.
        \item \textbf{LLM modules} for interpretation and constrained config refinement.
        \item \textbf{Graph ML} for anomaly and novelty detection in dependencies.
        \item Start from \textbf{OWASP AMASS}-style workflows and extend incrementally.
        \item Output: prioritized, explainable recommendations with provenance.
    \end{itemize}
    \note{This is orchestration and coordination, not replacement of analysts or existing tools.}
\end{frame}

        \draw[->,thick,dashed] (analyst) -- (policy);
    \end{tikzpicture}

    \vspace{0.2em}
    \begin{itemize}
        \item Sequential process under partial observability and non-determinism.
        \item Deterministic wrappers keep execution stable and replayable.
    \end{itemize}
    \note{This loop is the first thing we can ship as a baseline and iterate from.}
\end{frame}

\begin{frame}{Shared representation layer}
    \begin{itemize}
        \item Hybrid state: symbolic graph + learned embeddings.
        \item Entities: assets, services, endpoints, certs, identities, dependencies.
        \item Each observation stores provenance: source tool, config, time, confidence.
        \item LLM modules parse outputs and propose constrained refinements.
        \item Graph ML modules score anomaly and novelty for prioritization.
        \item One schema for orchestration, analyst review, and evaluation.
    \end{itemize}
    \note{This shared data layer is the core integration point across tools, models, and analyst decisions.}
\end{frame}

\section{Validation}
\begin{frame}{How we measure contributions value (and main risks)}
    \textbf{Value metrics}
    \begin{itemize}
        \item Discovery efficiency: relevant exposures per analyst-hour.
        \item Coverage and novelty quality on dependency graphs.
        \item Prioritization utility for analyst actionability.
        \item Explainability and traceability quality.
    \end{itemize}

    \textbf{Main risks and mitigations}
    \begin{itemize}
        \item RL instability \(\rightarrow\) staged baselines, curriculum, offline warm-start.
        \item LLM non-determinism \(\rightarrow\) constrained outputs, schema checks.
        \item Sim-to-real gap \(\rightarrow\) progressive validation with real tool behavior.
        \item Tool/API drift \(\rightarrow\) adapter contracts and regression tests.
    \end{itemize}
    \note{Success criteria and risk management are coupled; we track both from the first prototype.}
\end{frame}
