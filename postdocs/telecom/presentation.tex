\section{Proposed Research Direction}
\begin{frame}{Vision: Composable and Verifiable World Models}
    \justifying
    \textbf{Core idea:} Split the world model into modules with typed interfaces.
    \begin{block}{Modules}
        \begin{itemize}
            \item \textbf{Perception:} latent observation encoder + OOD detection.
            \item \textbf{Dynamics:} latent transition model (RSSM-like) + uncertainty estimation.
            \item \textbf{Constraints:} differentiable logic or hard safety guards.
        \end{itemize}
    \end{block}
    \textbf{Planning:} MPC / policy trained on model rollouts with constraint enforcement.
\end{frame}

% --------------------------------------------------------
\begin{frame}{Two Short-Term Research Axes}
    \justifying
    \begin{enumerate}
        \item \textbf{Constraint-Aware Dreamer:}
              \begin{itemize}
                  \item Integrate safety constraints and logic layers into latent model.
                  \item Evaluate on Safety-Gymnasium / DMControl.
              \end{itemize}
        \item \textbf{Graph-Constraint World Models:}
              \begin{itemize}
                  \item Represent dynamics as Graph Neural Network.
                  \item Constraints as graph predicates for compositional verification.
              \end{itemize}
    \end{enumerate}
    \vspace{0.2cm}
    Deliverables: open-source code, safety metrics, publication at NeurIPS/ICLR/ICML.
\end{frame}
