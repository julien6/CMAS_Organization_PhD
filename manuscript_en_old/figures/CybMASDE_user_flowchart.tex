\tikzset{
    startstop/.style={ellipse, draw, align=center, minimum width=2.5cm, minimum height=1cm},
    process/.style={rectangle, draw, align=center, minimum width=3.5cm, minimum height=1cm},
    decision/.style={diamond, draw, align=center, aspect=2, minimum width=3cm, minimum height=1cm},
    arrow/.style={-Latex, thick}
}

\begin{tikzpicture}[node distance=1.6cm]

    % Nodes
    \node (start) [startstop] {Début \\ Nouvel environnement inconnu};

    \node (create) [process, below=of start] {Création projet \\ Arborescence + JSON};

    \node (init) [decision, below=of create] {Fournir éléments initiaux ?};

    \node (api) [process, right=4.5cm of init] {Configurer API réelle};
    \node (manual) [process, below left=1.2cm and 4.5cm of init] {Environnement Python manuel};
    \node (traces) [process, below right=1.2cm and 4.5cm of init] {Collecte de traces initiales};

    \node (mod) [process, below=3.8cm of init] {Modelling (MOD) \\ World Model ou env. manuel};

    \node (train) [process, below=of mod] {Training (TRN) \\ MARL + Contraintes MOISE+MARL};

    \node (check) [decision, below=of train] {Récompense moyenne > seuil ?};

    \node (save) [process, right=4.5cm of check] {Sauvegarde politique \\ Déploiement Transferring};

    \node (analyze) [process, left=4.5cm of check] {Analyzing (ANL) \\ Auto-TEMM, rôles + objectifs};

    \node (deploy) [decision, below=of save] {Performance et stabilité satisfaisantes ?};

    \node (end) [startstop, below=of deploy] {Fin \\ Politique conjointe validée \\ + Statistiques + Specs organisationnelles};

    \draw[arrow] (start) -- (create);

    \draw[arrow] (api.south) |- (mod.east);

    \draw[arrow] (traces.south) |- (mod.east);

    \draw[arrow] (mod) -- (train);
    \draw[arrow] (train) -- (check);

    \draw[arrow] (check.east) -- ++(1.5,0) |- (save.north) node[midway, above] {Oui};
    \draw[arrow] (check.west) -- ++(-1.5,0) |- (analyze.north) node[midway, above] {Non};

    \draw[arrow] (analyze) -- (refine);
    \draw[arrow] (refine.east) -- ++(1.5,0) |- (train.west);

    \draw[arrow] (save) -- (deploy);
    \draw[arrow] (deploy) -- (end) node[midway, right] {Oui};
    \draw[arrow] (deploy.west) -- ++(-1.5,0) |- (analyze.south) node[midway, below] {Non};
